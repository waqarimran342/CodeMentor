// ========== FILE HANDLING ==========

/// Q1
#include <fstream>
int main() {
    ofstream myfile;
    myfile.open("data.txt")
    myfile << "Hello";
    myfile.close();
}

/// Q2
#include <fstream>
int main() {
    ifstream file("info.txt");
    string data;
    file >> data;
    cout << data;
}

}

    ofstream file("log.txt");
    if (!file) {
        cout << "File failed";
    }
    file << "Log entry";
}

    ifstream inFile;
