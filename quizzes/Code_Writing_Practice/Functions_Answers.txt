/* ********** DIFFICULT QUESTION 1 ********** */

#include <iostream>
using namespace std;
int factorial(int n) {
    int f = 1;
    for(int i = 1; i <= n; i++) f *= i;
    return f;
}
int main() {
    cout << factorial(5);
    return 0;
}

/* ********** DIFFICULT QUESTION 2 ********** */

#include <iostream>
using namespace std;
bool isEven(int x) {
    return x % 2 == 0;
}
int main() {
    int num;
    cin >> num;
    cout << (isEven(num) ? "Even" : "Odd");
    return 0;
}

/* ********** DIFFICULT QUESTION 3 ********** */

#include <iostream>
using namespace std;
int maxNum(int a, int b) {
    return (a > b) ? a : b;
}
int main() {
    int x = 10, y = 20;
    cout << "Maximum = " << maxNum(x, y);
    return 0;
}
