/* ********** DIFFICULT QUESTION 2 ********** */

#include <iostream>
#include <fstream>
using namespace std;
int main() {
    ofstream fout("numbers.txt");
    for(int i = 1; i <= 10; i++) fout << i << " ";
    fout.close();

    ifstream fin("numbers.txt");
    int x, sum = 0;
    while(fin >> x) sum += x;
    cout << "Sum = " << sum;
    fin.close();
    return 0;
}
