/* ************ EASY QUESTION 1 ************ */

#include <iostream>
using namespace std;
int main() {
    int day;
    cout << "Enter day number (1-7): ";
    cin >> day;
    switch(day) {
        case 1: cout << "Monday"; break;
        case 2: cout << "Tuesday"; break;
        case 3: cout << "Wednesday"; break;
        case 4: cout << "Thursday"; break;
        case 5: cout << "Friday"; break;
        case 6: cout << "Saturday"; break;
        case 7: cout << "Sunday"; break;
        default: cout << "Invalid";
    }
    return 0;
}

/* ************ EASY QUESTION 2 ************ */

#include <iostream>
using namespace std;
int main() {
    char op;
    float a, b;
    cout << "Enter operator (+ - * /): ";
    cin >> op;
    cout << "Enter two numbers: ";
    cin >> a >> b;
    switch(op) {
        case '+': cout << a + b; break;
        case '-': cout << a - b; break;
        case '*': cout << a * b; break;
        case '/': cout << a / b; break;
        default: cout << "Invalid";
    }
    return 0;
}
