/* ************ INPUT & OUTPUT ************ */

#include <iostream>
using namespace std;

int main() {
    int age;
    cout << "Enter your age: ";
    cin >> age;
    cout << "You are " << age << " years old.";
    return 0;
}


/* *************** IF-ELSE *************** */

int num = 5;
if (num > 0) {
    cout << "Positive";
} else if (num < 0) {
    cout << "Negative";
} else {
    cout << "Zero";
}


/* *************** SWITCH *************** */

int day = 2;
switch(day) {
    case 1: cout << "Monday"; break;
    case 2: cout << "Tuesday"; break;
    default: cout << "Other day";
}


/* *************** LOOPS *************** */

// For loop
for (int i = 0; i < 5; i++) {
    cout << i << " ";
}

// While loop
int i = 0;
while (i < 5) {
    cout << i << " ";
    i++;
}

// Do-while loop
int j = 0;
do {
    cout << j << " ";
    j++;
} while (j < 5);


/* ************** FUNCTIONS ************** */

int add(int a, int b) {
    return a + b;
}

int main() {
    cout << add(3, 4);
}


/* *************** ARRAYS *************** */

int arr[3] = {10, 20, 30};
for(int i = 0; i < 3; i++) {
    cout << arr[i] << " ";
}


/* ************** POINTERS ************** */

int x = 10;
int* ptr = &x;
cout << "Value: " << *ptr;


/* ************** STRUCTS *************** */

struct Student {
    string name;
    int age;
};

Student s1 = {"Ali", 20};
cout << s1.name;


/* ************ FILE HANDLING ************ */
