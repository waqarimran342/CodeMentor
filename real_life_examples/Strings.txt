/************************************************************
*  EXPLANATION:                                            *
*  Strings are used to handle text. The string class in   *
*  C++ provides easy ways to store, concatenate, and      *
*  measure text. Useful for names, messages, or any text  *
*  processing.                                            *
************************************************************/


/*************** EXAMPLE 1: CONCATENATE FIRST AND LAST NAMES ***************/

#include <iostream>
#include <string>
using namespace std;

int main() {
    string firstName, lastName;
    cout << "Enter first name: ";
    cin >> firstName;
    cout << "Enter last name: ";
    cin >> lastName;
    string fullName = firstName + " " + lastName;
    cout << "Full Name: " << fullName;
    return 0;
}


/*************** EXAMPLE 2: FIND LENGTH OF A STRING ***************/

#include <iostream>
#include <string>
using namespace std;

int main() {
    string word;
    cout << "Enter a word: ";
    cin >> word;
    cout << "Length: " << word.length();
    return 0;
}
