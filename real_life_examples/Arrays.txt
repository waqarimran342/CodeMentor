/****************************************************/

/************************************************************
*  EXPLANATION:                                            *
*  Arrays allow you to store MULTIPLE values of the SAME  *
*  DATA TYPE under ONE variable name. This is handy for   *
*  handling collections of data like student marks or     *
*  daily temperatures efficiently, instead of using many *
*  separate variables.                                     *
************************************************************/


/***************** EXAMPLE 1: AVERAGE MARKS *****************/

#include <iostream>
using namespace std;

int main() {
    int marks[5];            // Array to hold marks for 5 students
    int sum = 0;

    cout << "Enter marks of 5 students:\n";
    for (int i = 0; i < 5; i++) {
        cin >> marks[i];     // Read marks one at a time
        sum += marks[i];     // Keep adding to sum
    }

    cout << "Average marks: " << (sum / 5.0) << endl;  // Calculate & display average
    return 0;
}

/************ EXAMPLE 2: MAXIMUM TEMPERATURE IN A WEEK ************/

#include <iostream>
using namespace std;

int main() {
    int temps[7] = {30, 32, 31, 35, 33, 29, 28};  // Temperatures for 7 days
    int maxTemp = temps[0];                        // Assume first day temp is max initially

    for (int i = 1; i < 7; i++) {
        if (temps[i] > maxTemp) {
            maxTemp = temps[i];                    // Update maxTemp if current day is hotter
        }
    }

    cout << "Maximum temperature this week: " << maxTemp << "°C" << endl;
    return 0;
}

/*                    CASE_SWITCH                   */
/****************************************************/

/************************************************************
*  EXPLANATION:                                            *
*  Switch statements allow you to handle multiple        *
*  discrete choices efficiently. Instead of many if-else *
*  checks, switch keeps code clean and readable when      *
*  dealing with options like menu selection or operator   *
*  choice.                                                *
************************************************************/


/***************** EXAMPLE 1: SIMPLE CALCULATOR *****************/

#include <iostream>
using namespace std;

int main() {
    char op;
    double num1, num2;

    cout << "Enter operator (+, -, *, /): ";
    cin >> op;

    cout << "Enter two numbers: ";
    cin >> num1 >> num2;

    switch(op) {
        case '+': cout << num1 + num2; break;
        case '-': cout << num1 - num2; break;
        case '*': cout << num1 * num2; break;
        case '/':
            if (num2 != 0) 
                cout << num1 / num2;
            else 
                cout << "Cannot divide by zero";
            break;
        default:
            cout << "Invalid operator";
    }
    return 0;
}


/************ EXAMPLE 2: MENU SELECTION IN GAME/APP ************/

#include <iostream>
using namespace std;

int main() {
    int choice;

    cout << "Menu:\n1. Play\n2. Settings\n3. Exit\nChoose option: ";
    cin >> choice;

    switch(choice) {
        case 1: cout << "Game started!"; break;
        case 2: cout << "Settings opened!"; break;
        case 3: cout << "Exiting..."; break;
        default: cout << "Invalid choice";
    }
    return 0;
}
